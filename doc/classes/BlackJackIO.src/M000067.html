  def show_hands(players)
    if players.is_a?(Array)
      for player in players
        for hand in player.hands
          puts "#{player.hand_to_s(hand)}: #{hand.to_s}"
        end
      end
    else
      player = players
      for hand in player.hands
        puts "#{player.hand_to_s(hand)}: #{hand.to_s}"
      end
    end
  end
