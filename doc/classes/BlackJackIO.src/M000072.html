  def want_to_play(shoe)
    result = nil
    while not result
      print "\nShoe #{shoe} finished.  Want to play another shoe [y|n]? "
      result = gets
      result = result.strip.downcase
      case result
      when "y"
        return 1
      when "n", "q"
        throw :quit
      when "i"
        instructions
      end
      result = nil
      try_again
    end
  end
