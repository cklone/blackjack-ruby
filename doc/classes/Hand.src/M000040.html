  def to_s
    result = "[#{@cards.join("|")}]: "
    if is_soft
      result += total.join(" or ")
    else
      result += total.to_s
    end
    if is_bust
      result += " BUSTED"
    elsif is_bj
      result += " BLACKJACK"
    elsif total_high == 21
      result += " NICE"
    end
    return result
  end
