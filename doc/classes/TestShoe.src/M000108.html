  def test_shoe_creation
    1.upto(8) do |x|
      shoe = Shoe.new(x)
      assert_equal(52 * x, shoe.cards.length, "Shoe card counts [#{x}]")
      assert_not_nil(shoe, "Shoe created with #{x} decks.")
      assert_nothing_raised(ArgumentError) { Shoe.new(x) }
    end
  end
